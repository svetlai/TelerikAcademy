8. Write a script that converts a number in the range [0...999] to a text corresponding to its English pronunciation. 


function numberName(number) {
    var number = parseInt(document.getElementById("a8").value),
        digit1 = number % 10,
        digit10 = parseInt(number / 10) % 10,
        digit100 = parseInt(number / 100) % 10;

    function returnDigitName(number) {
        var digName = "";

        switch (number) {
            case 1: digName = "One"; break;
            case 2: digName = "Two"; break;
            case 3: digName = "Three"; break;
            case 4: digName = "Four"; break;
            case 5: digName = "Five"; break;
            case 6: digName = "Six"; break;
            case 7: digName = "Seven"; break;
            case 8: digName = "Eight"; break;
            case 9: digName = "Nine"; break;
        }
        return digName;
    }
    if (isNaN(number)) {
        console.log("This is not a number.");
    } else if (number > 999 || number < 0) {
        console.log("Type a number between 0 & 999")
    } else {
        if (digit1 === 0 && digit10 === 0 && digit100 === 0) {      //zero
            console.log("Zero");
        }
        else if (digit10 === 0 && digit100 === 0) {                 // 1 - 10
            console.log(returnDigitName(digit1));
        }
        if (digit100 !== 0) {
            console.log(returnDigitName(digit100) + " Hundred ");       //hundreds
            if (digit1 !== 0 || digit10 !== 0) {
                console.log("and ")
            }
        }
        if (digit10 !== 0) {                                        //tens
            if (digit10 === 1) {
                switch (digit1) {                                   //10 - 19
                    case 0: console.log("Ten"); break;
                    case 1: console.log("Eleven"); break;
                    case 2: console.log("Twelve"); break;
                    case 3: console.log("Thirteen"); break;
                    case 5: console.log("Fifteen"); break;
                    case 8: console.log("Eighteen"); break;
                    default: console.log(returnDigitName(digit1) + "teen"); break;
                }
            }
           else {
                    switch (digit10) {                          //20-90
                        case 2: console.log("Twenty"); break;
                        case 3: console.log("Thirty"); break;
                        case 5: console.log("Fifty"); break;
                        case 8: console.log("Eighty"); break;
                        default: console.log(returnDigitName(digit10) + "ty"); break;
                    }
            }
        }
        if (digit1 !== 0 && digit10 !== 1 && digit100 !== 0 || digit1 !== 0 && digit10 !== 1 && digit10 !== 0)    //adds last digit if over 20
        {
            console.log(returnDigitName(digit1));
        }
    }
